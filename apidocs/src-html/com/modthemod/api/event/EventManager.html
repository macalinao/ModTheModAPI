001    package com.modthemod.api.event;
002    
003    /**
004     * Manages {@link Event}s and {@link EventType}s.
005     */
006    public interface EventManager {
007            /**
008             * Registers an {@link EventType} with the game.
009             * 
010             * @param event
011             *            The {@link EventType} to register.
012             */
013            public void registerEvent(EventType type);
014    
015            /**
016             * Gets the {@link EventType} corresponding with the given name.
017             * 
018             * @param name
019             *            The name of the {@link EventType}.
020             * @return The {@link EventType} corresponding with the given name.
021             */
022            public EventType getEvent(String name);
023    }
