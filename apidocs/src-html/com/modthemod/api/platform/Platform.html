001    package com.modthemod.api.platform;

003    /**
004     * Represents a platform the game is running on, for example, Mojang API,
005     * Terasology, or Spout.
006     */
007    public interface Platform {
008            /**
009             * Gets the name of the {@link Platform}.
010             * 
011             * @return The name of the {@link Platform}.
012             */
013            public String getName();
014    
015            /**
016             * Gets the version of the {@link Platform}.
017             * 
018             * @return The version of the {@link Platform}.
019             */
020            public String getVersion();
021    }
