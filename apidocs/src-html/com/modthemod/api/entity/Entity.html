001    package com.modthemod.api.entity;
002    
003    import com.modthemod.api.base.Base;
004    import com.modthemod.api.property.Type;
005    
006    /**
007     * Represents an {@link Entity}, or a tangible object in-game.
008     */
009    public abstract class Entity {
010            /**
011             * The {@link Base} of the {@link Entity}.
012             */
013            private final Base base;
014    
015            /**
016             * Creates a new {@link Entity} with the given base.
017             * 
018             * <p>
019             * <strong>NOTE: Please use {@link Base#spawn} instead.</strong>
020             * </p>
021             * 
022             * @param base
023             *            The {@link Base} of the entity.
024             */
025            public Entity(Base base) {
026                    this.base = base;
027            }
028    
029            /**
030             * Gets the {@link Base} of the {@link Entity.}
031             * 
032             * @return The {@link Base} of the {@link Entity.
033             */
034            public Base getBase() {
035                    return base;
036            }
037    
038            /**
039             * Gets the value of a property from its name.
040             * 
041             * @param name
042             *            The name of the property.
043             * @return The value of the property as a {@link Type}.
044             */
045            public abstract Type<?> getProperty(String name);
046    
047            /**
048             * Sets the value of a property from its name.
049             * 
050             * @param name
051             *            The name of the property.
052             * @param value
053             *            The value of the property as a {@link Type}.
054             */
055            public abstract void setProperty(String name, Type<?> value);
056    }
