001    package com.modthemod.api.entity;
002    
003    import java.util.Map;
004    
005    import com.modthemod.api.base.Base;
006    import com.modthemod.api.property.Property;
007    
008    /**
009     * Represents an {@link Entity}, or a tangible object in-game.
010     */
011    public abstract class Entity {
012            /**
013             * The {@link Base} of the {@link Entity}.
014             */
015            private final Base base;
016    
017            /**
018             * Creates a new {@link Entity} with the given base.
019             * 
020             * <p>
021             * <strong>NOTE: Please use {@link Base#spawn} instead.</strong>
022             * </p>
023             * 
024             * @param base
025             *            The {@link Base} of the entity.
026             */
027            public Entity(Base base) {
028                    this.base = base;
029            }
030    
031            /**
032             * Gets the value of a property from its name.
033             * 
034             * @param name
035             *            The name of the property.
036             * @return The value of the property as a {@link Property}.
037             */
038            public abstract Property getProperty(String name);
039    
040            /**
041             * Sets the value of a property from its name.
042             * 
043             * @param name
044             *            The name of the property.
045             * @param value
046             *            The value of the property as a {@link Property}.
047             */
048            public abstract void setProperty(String name, Property value);
049    }
