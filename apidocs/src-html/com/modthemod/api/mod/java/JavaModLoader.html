001    package com.modthemod.api.mod.java;

003    import java.io.File;
004    import java.util.Arrays;
005    import java.util.List;
006    import java.util.regex.Pattern;
007    
008    import com.modthemod.api.mod.ModDescription;
009    import com.modthemod.api.mod.ModLoader;
010    
011    public class JavaModLoader implements ModLoader<JavaMod> {
012            public static final List<Pattern> FILTERS = Arrays.asList(Pattern
013                            .compile("\\.jar$"));
014    
015            @Override
016            public String getLanguage() {
017                    return "Java";
018            }
019    
020            @Override
021            public JavaMod loadMod(File file) {
022                    return new JavaMod(); // TODO
023            }
024    
025            @Override
026            public ModDescription loadDescription(File file) {
027                    return new ModDescription(); // TODO
028            }
029    
030            @Override
031            public List<Pattern> getFileFilters() {
032                    return FILTERS;
033            }
034    
035            @Override
036            public void enableMod(JavaMod mod) {
037                    if (!mod.isEnabled()) {
038                            mod.setEnabled(true);
039                    }
040            }
041    
042            @Override
043            public void disableMod(JavaMod mod) {
044                    if (mod.isEnabled()) {
045                            mod.setEnabled(false);
046                    }
047            }
048    
049    }
