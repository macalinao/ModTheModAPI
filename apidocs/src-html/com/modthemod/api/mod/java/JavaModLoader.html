001    package com.modthemod.api.mod.java;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.io.InputStream;
006    import java.lang.reflect.Constructor;
007    import java.net.URL;
008    import java.util.Arrays;
009    import java.util.HashMap;
010    import java.util.List;
011    import java.util.Map;
012    import java.util.jar.JarEntry;
013    import java.util.jar.JarFile;
014    import java.util.regex.Pattern;
015    
016    import com.modthemod.api.Game;
017    import com.modthemod.api.mod.InvalidModException;
018    import com.modthemod.api.mod.ModDescription;
019    import com.modthemod.api.mod.ModLoader;
020    
021    public class JavaModLoader implements ModLoader<JavaMod> {
022            /**
023             * Filters for getting jars.
024             */
025            public static final List<Pattern> FILTERS = Arrays.asList(Pattern
026                            .compile("\\.jar$"));
027    
028            /**
029             * The game, which you just lost.
030             */
031            private final Game game;
032    
033            /**
034             * The classes which are important.
035             */
036            protected final Map<String, Class<?>> classes = new HashMap<String, Class<?>>();
037    
038            /**
039             * The class loaders we are using.
040             */
041            protected final Map<String, ModClassLoader> loaders = new HashMap<String, ModClassLoader>();
042    
043            /**
044             * Constructor.
045             * 
046             * @param game
047             *            The game.
048             */
049            public JavaModLoader(Game game) {
050                    this.game = game;
051            }
052    
053            @Override
054            public String getLanguage() {
055                    return "Java";
056            }
057    
058            @Override
059            public JavaMod loadMod(File file) throws InvalidModException {
060                    JavaMod result = null;
061                    ModDescription desc = null;
062    
063                    desc = loadDescription(file);
064    
065                    File dataFolder = new File(game.getPlatform().getFileHierarchy()
066                                    .getDataFolder(), desc.getName());
067    
068                    ModClassLoader classLoader = new ModClassLoader(this, getClass()
069                                    .getClassLoader());
070                    try {
071                            URL[] urls = new URL[1];
072                            urls[0] = file.toURI().toURL();
073    
074                            Class<?> main = Class.forName(desc.getMain(), true, classLoader);
075                            Class<? extends JavaMod> mod = main.asSubclass(JavaMod.class);
076                            Constructor<? extends JavaMod> constructor = mod.getConstructor();
077    
078                            result = constructor.newInstance();
079                            result.initialize(dataFolder, desc, this);
080    
081                    } catch (Exception e) {
082                            throw new InvalidModException(e);
083                    }
084    
085                    loaders.put(desc.getName(), classLoader);
086                    return result;
087            }
088    
089            /**
090             * Loads the description of the given plugin file.
091             * 
092             * @param file
093             * @return
094             * @throws InvalidModException
095             */
096            private ModDescription loadDescription(File file)
097                            throws InvalidModException {
098                    JarFile jar = null;
099                    InputStream stream = null;
100                    ModDescription desc = null;
101                    try {
102                            jar = new JarFile(file);
103                            JarEntry entry = jar.getJarEntry("mod.json");
104    
105                            if (entry == null) {
106                                    throw new InvalidModException("mod.json does not exist!");
107                            }
108    
109                            stream = jar.getInputStream(entry);
110                            desc = ModDescription.load(file);
111                    } catch (IOException ex) {
112                            throw new InvalidModException(ex);
113                    } finally {
114                            if (stream != null) {
115                                    try {
116                                            stream.close();
117                                    } catch (IOException e) {
118                                            e.printStackTrace();
119                                    }
120                            }
121                            if (jar != null) {
122                                    try {
123                                            jar.close();
124                                    } catch (IOException e) {
125                                            e.printStackTrace();
126                                    }
127                            }
128                    }
129    
130                    return desc;
131            }
132    
133            @Override
134            public List<Pattern> getFileFilters() {
135                    return FILTERS;
136            }
137    
138            @Override
139            public void enableMod(JavaMod mod) {
140                    if (!mod.isEnabled()) {
141                            mod.setEnabled(true);
142                    }
143            }
144    
145            @Override
146            public void disableMod(JavaMod mod) {
147                    if (mod.isEnabled()) {
148                            mod.setEnabled(false);
149                    }
150            }
151

153                    // TODO Auto-generated method stub
154                    return null;
155            }
156    
157            public void setClass(String name, Class<?> result) {
158                    // TODO Auto-generated method stub
159    
160            }
161    
162    }
