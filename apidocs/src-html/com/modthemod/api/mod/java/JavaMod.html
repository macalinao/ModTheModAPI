001    package com.modthemod.api.mod.java;
002    
003    import java.util.logging.Logger;
004    
005    import com.modthemod.api.mod.Mod;
006    import com.modthemod.api.mod.ModDescription;
007    import com.modthemod.api.mod.ModLoader;
008    import com.modthemod.api.mod.ModLogger;
009    
010    /**
011     * A Java mod.
012     */
013    public class JavaMod implements Mod {
014            private ModLoader loader;
015    
016            private Logger logger;
017    
018            private boolean initialized = false;
019    
020            /**
021             * The description for the mod.
022             */
023            private ModDescription description;
024    
025            /**
026             * Called when the mod is enabled.
027             */
028            public void onEnable() {
029            }
030    
031            /**
032             * Called when the mod is disabled.
033             */
034            public void onDisable() {
035            }
036    
037            @Override
038            public String getName() {
039                    return getDescription().getName();
040            }
041    
042            @Override
043            public ModDescription getDescription() {
044                    return description;
045            }
046    
047            @Override
048            public Logger getLogger() {
049                    return logger;
050            }
051    
052            public ModLoader getModLoader() {
053                    return loader;
054            }
055    
056            /**
057             * Initializes the mod. Basically gets it ready for being enabled.
058             * 
059             * @param loader
060             */
061            void initialize(ModDescription description, ModLoader loader) {
062                    if (initialized) {
063                            throw new IllegalStateException("Mod already initialized!");
064                    }
065    
066                    this.description = description;
067                    this.loader = loader;
068                    this.logger = new ModLogger(this);
069    
070                    initialized = true;
071            }
072    
073    }
