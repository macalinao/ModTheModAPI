001    package com.modthemod.api.mod.java;
002    
003    import java.net.URL;
004    import java.net.URLClassLoader;
005    import java.util.HashMap;
006    import java.util.Map;
007    import java.util.Set;
008    
009    /**
010     * An awesome class loader. It's awesome.
011     */
012    public class ModClassLoader extends URLClassLoader {
013            private final JavaModLoader loader;
014    
015            private final Map<String, Class<?>> classes = new HashMap<String, Class<?>>();
016    
017            public ModClassLoader(final JavaModLoader loader,
018                            final ClassLoader parent) {
019                    super(new URL[0], parent);
020                    this.loader = loader;
021            }
022    
023            @Override
024            protected void addURL(URL url) {
025                    super.addURL(url);
026            }
027    
028            @Override
029            protected Class<?> findClass(String name) throws ClassNotFoundException {
030                    return findClass(name, true);
031            }
032    
033            /**
034             * Find a class.
035             * 
036             * @param name
037             * @param checkGlobal
038             * @return
039             * @throws ClassNotFoundException
040             */
041            protected Class<?> findClass(String name, boolean checkGlobal)
042                            throws ClassNotFoundException {
043                    Class<?> result = classes.get(name);
044    
045                    if (result == null) {
046                            if (checkGlobal) {
047                                    result = loader.getClassByName(name);
048                            }
049    
050                            if (result == null) {
051                                    result = super.findClass(name);
052    
053                                    if (result != null) {
054                                            loader.setClass(name, result);
055                                    }
056                            }
057    
058                            classes.put(name, result);
059                    }
060    
061                    return result;
062            }
063    
064            public Set<String> getClasses() {
065                    return classes.keySet();
066            }
067    }
