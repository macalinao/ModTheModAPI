001    package com.modthemod.api.mod;
002    
003    import java.util.Set;
004    
005    /**
006     * Manages mods.
007     */
008    public interface ModManager {
009            /**
010             * Gets the mod with the given name.
011             * 
012             * @param name
013             * @return
014             */
015            public Mod getMod(String name);
016    
017            /**
018             * Gets a set of all loaded mods.
019             * 
020             * @return
021             */
022            public Set<Mod> getMods();
023    
024            /**
025             * Registers a {@link ModLoader} with the {@link ModManager}.
026             * 
027             * @param loader
028             */
029            public void registerModLoader(ModLoader<?> loader);
030    }
