001    package com.modthemod.api.mod;
002    
003    import java.io.File;
004    import java.util.List;
005    import java.util.regex.Pattern;
006    
007    /**
008     * Represents an interface to load {@link Mod}s.
009     */
010    public interface ModLoader<T extends Mod> {
011            /**
012             * Gets the language of the mods loaded by this {@link ModLoader}.
013             * 
014             * @return The language of the mods loaded by this {@link ModLoader}.
015             */
016            public String getLanguage();
017    
018            /**
019             * Loads a {@link Mod} with the given {@link File}.
020             * 
021             * @param file
022             *            The {@link File} to load.
023             * @return The {@link Mod} loaded.
024             */
025            public T loadMod(File file);
026    
027            /**
028             * Loads a {@link ModDescription} from its {@link File}.
029             * 
030             * @param file
031             *            The {@link File} containing the {@link ModDescription}. (Read:
032             *            mod.json)
033             * @return The parsed {@link ModDescription}.
034             */
035            public ModDescription loadDescription(File file);
036    
037            /**
038             * Gets a {@link List} of all file filters that match files of this
039             * {@link ModLoader}'s type.
040             * 
041             * @return A {@link List} of all file filters that match files of this
042             *         {@link ModLoader}'s type.
043             */
044            public List<Pattern> getFileFilters();
045    
046            /**
047             * Enables the {@link Mod}.
048             * 
049             * @param mod
050             *            The {@link Mod} to enable.
051             */
052            public void enableMod(T mod);
053    
054            /**
055             * Disables the {@link Mod}.
056             * 
057             * @param mod
058             *            The {@link Mod} to disable.
059             */
060            public void disableMod(T mod);
061    }
