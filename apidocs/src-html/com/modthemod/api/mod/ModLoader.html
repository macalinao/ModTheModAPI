001    package com.modthemod.api.mod;
002    
003    import java.io.File;
004    import java.util.List;
005    import java.util.regex.Pattern;
006    
007    /**
008     * Represents an interface to load {@link Mod}s.
009     */
010    public interface ModLoader {
011            /**
012             * Gets the language of the mods loaded by this {@link ModLoader}.
013             * 
014             * @return The language of the mods loaded by this {@link ModLoader}.
015             */
016            public String getLanguage();
017    
018            /**
019             * Loads a {@link Mod} with the given {@link File}.
020             * 
021             * @param file
022             *            The {@link File} to load.
023             * @return The {@link Mod} loaded.
024             */
025            public Mod loadMod(File file);
026    
027            /**
028             * Gets a {@link List} of all file filters that match files of this
029             * {@link ModLoader}'s type.
030             * 
031             * @return A {@link List} of all file filters that match files of this
032             *         {@link ModLoader}'s type.
033             */
034            public List<Pattern> getFileFilters();
035    
036            /**
037             * Enables the {@link Mod}.
038             * 
039             * @param mod
040             *            The {@link Mod} to enable.
041             */
042            public void enableMod(Mod mod);
043    
044            /**
045             * Disables the {@link Mod}.
046             * 
047             * @param mod
048             *            The {@link Mod} to disable.
049             */
050            public void disableMod(Mod mod);
051    }
