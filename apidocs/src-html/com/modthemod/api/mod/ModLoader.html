001    package com.modthemod.api.mod;
002    
003    import java.io.File;
004    import java.util.List;
005    import java.util.regex.Pattern;
006    
007    /**
008     * Represents an interface to load {@link Mod}s.
009     */
010    public interface ModLoader {
011            /**
012             * Loads a {@link Mod} with the given {@link File}.
013             * 
014             * @param file
015             *            The {@link File} to load.
016             * @return The {@link Mod} loaded.
017             */
018            public Mod loadMod(File file);
019    
020            /**
021             * Gets a {@link List} of all file filters that match files of this
022             * {@link ModLoader}'s type.
023             * 
024             * @return A {@link List} of all file filters that match files of this
025             *         {@link ModLoader}'s type.
026             */
027            public List<Pattern> getFileFilters();
028    
029            /**
030             * Enables the {@link Mod}.
031             * 
032             * @param mod
033             *            The {@link Mod} to enable.
034             */
035            public void enableMod(Mod mod);
036    
037            /**
038             * Disables the {@link Mod}.
039             * 
040             * @param mod
041             *            The {@link Mod} to disable.
042             */
043            public void disableMod(Mod mod);
044    }
