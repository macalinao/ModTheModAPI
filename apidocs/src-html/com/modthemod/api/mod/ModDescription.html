001    package com.modthemod.api.mod;
002    
003    import java.io.File;
004    import java.io.FileInputStream;
005    import java.io.FileNotFoundException;
006    import java.io.IOException;
007    
008    import com.google.gson.Gson;
009    
010    /**
011     * Contains the description for the mod.
012     */
013    public class ModDescription {
014            /**
015             * The name of the mod.
016             */
017            private String name;
018    
019            /**
020             * The version of the mod.
021             */
022            private String version;
023    
024            /**
025             * The author of the mod.
026             */
027            private String author;
028    
029            /**
030             * The description of the mod.
031             */
032            private String description;
033    
034            /**
035             * The authors to the mod. These are not as important as the main author.
036             */
037            private String[] authors;
038    
039            /**
040             * The main JS file of the mod.
041             */
042            private String main;
043    
044            /**
045             * Github link to the repository.
046             * 
047             * <p>
048             * Yes, you must use Github. Any other code sharing site is crap. If you
049             * don't put your work on Github, well, that's a different story.
050             * </p>
051             */
052            private String github;
053    
054            /**
055             * The dependencies of the mod. (mod-version, e.g. core-0.0.1)
056             */
057            private String[] dependencies;
058    
059            /**
060             * The license of the mod. The name of the license, that is. We don't want
061             * 10 copies of the AGPL floating around in memory.
062             */
063            private String license;
064    
065            /**
066             * The minimum version of ModTheMod that this is compatible with.
067             */
068            private String engineVersion;
069    
070            /**
071             * @return the name
072             */
073            public String getName() {
074                    return name;
075            }
076    
077            /**
078             * @return the version
079             */
080            public String getVersion() {
081                    return version;
082            }
083    
084            /**
085             * @return the author
086             */
087            public String getAuthor() {
088                    return author;
089            }
090    
091            /**
092             * @return the description
093             */
094            public String getDescription() {
095                    return description;
096            }
097    
098            /**
099             * @return the authors
100             */
101            public String[] getAuthors() {
102                    return authors;
103            }
104    
105            /**
106             * @return the main
107             */
108            public String getMain() {
109                    return main;
110            }
111    
112            /**
113             * @return the github
114             */
115            public String getGithub() {
116                    return github;
117            }
118    
119            /**
120             * @return the dependencies
121             */
122            public String[] getDependencies() {
123                    return dependencies;
124            }
125    
126            /**
127             * @return the license
128             */
129            public String getLicense() {
130                    return license;
131            }
132    
133            /**
134             * @return the engineVersion
135             */
136            public String getEngineVersion() {
137                    return engineVersion;
138            }
139    
140            /**
141             * Loads a {@link ModDescription} from its file.
142             * 
143             * @param file
144             *            The file.
145             * @return The {@link ModDescription}
146             * @throws IOException
147             *             If the file isn't loaded correctly.
148             */
149            public static ModDescription load(File file) throws IOException {
150                    int ch;
151                    StringBuilder json = new StringBuilder("");
152    
153                    FileInputStream fin = new FileInputStream(file);
154    
155                    while ((ch = fin.read()) != -1) {
156                            json.append((char) ch);
157                    }
158    
159                    fin.close();
160    
161                    return load(json.toString());
162            }
163    
164            /**
165             * Loads a {@link ModDescription} from its JSON representation.
166             * 
167             * @param json
168             *            The JSON representation.
169             * @return The {@link ModDescription}.
170             */
171            public static ModDescription load(String json) {
172                    ModDescription description = (new Gson()).fromJson(json,
173                                    ModDescription.class);
174    
175                    return description;
176            }
177    }
