001    package com.modthemod.api.mod;
002    
003    import java.util.logging.Logger;
004    
005    /**
006     * Represents a mod.
007     */
008    public interface Mod<T extends Mod<T>> {
009            /**
010             * Gets the name of this {@link Mod}.
011             * 
012             * @return The name of this {@link Mod}.
013             */
014            public String getName();
015    
016            /**
017             * Gets the {@link ModDescription} of this mod.
018             * 
019             * @return The {@link ModDescription} of this mod.
020             */
021            public ModDescription getDescription();
022    
023            /**
024             * Gets the {@link Logger} for this mod.
025             * 
026             * @return The {@link Logger} for this mod.
027             */
028            public Logger getLogger();
029    
030            /**
031             * Returns true if the {@link Mod} is enabled.
032             * 
033             * @return True if the {@link Mod} is enabled.
034             */
035            public boolean isEnabled();
036    
037            /**
038             * Called when the mod is loaded.
039             */
040            public void onLoad();
041    
042            /**
043             * Called when the mod is enabled.
044             */
045            public void onEnable();
046    
047            /**
048             * Called when the mod is disabled.
049             */
050            public void onDisable();
051    
052            /**
053             * Gets the {@link ModLoader} that loaded this mod.
054             * 
055             * @return The {@link ModLoader} that loaded this mod.
056             */
057            public ModLoader<T> getModLoader();
058    }
