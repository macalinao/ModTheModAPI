001    package com.modthemod.api.mod;
002    
003    /**
004     * Represents a mod.
005     */
006    public abstract class Mod {
007            private final String name;
008    
009            protected Mod(String name) {
010                    this.name = name;
011            }
012    
013            /**
014             * Gets the name of the mod.
015             * 
016             * @return The name of the mod.
017             */
018            public String getName() {
019                    return name;
020            }
021    }
