001    package com.modthemod.api.mod;
002    
003    import java.io.File;
004    
005    /**
006     * Contains the description for the mod.
007     */
008    public class ModDescription {
009            /**
010             * The name of the mod.
011             */
012            private String name;
013    
014            /**
015             * The version of the mod.
016             */
017            private String version;
018    
019            /**
020             * The author of the mod.
021             */
022            private String author;
023    
024            /**
025             * The description of the mod.
026             */
027            private String description;
028    
029            /**
030             * The {link Contributor}s to the mod.
031             */
032            private Contributor[] contributors;
033    
034            /**
035             * The main JS file of the mod.
036             */
037            private String main;
038    
039            /**
040             * Github link to the repository.
041             * 
042             * <p>
043             * Yes, you must use Github. Any other code sharing site is crap. If you
044             * don't put your work on Github, well, that's a different story.
045             * </p>
046             */
047            private String github;
048    
049            /**
050             * The dependencies of the mod. (mod-version, e.g. core-0.0.1)
051             */
052            private String[] dependencies;
053    
054            /**
055             * The license of the mod. The name of the license, that is. We don't want
056             * 10 copies of the AGPL floating around in memory.
057             */
058            private String license;
059    
060            /**
061             * The minimum version of ModTheMod that this is compatible with.
062             */
063            private String engineVersion;
064    
065            public static ModDescription load(File file) {
066                    return null; // TODO
067            }
068    
069            /**
070             * @return the name
071             */
072            public String getName() {
073                    return name;
074            }
075    
076            /**
077             * @return the version
078             */
079            public String getVersion() {
080                    return version;
081            }
082    
083            /**
084             * @return the author
085             */
086            public String getAuthor() {
087                    return author;
088            }
089    
090            /**
091             * @return the description
092             */
093            public String getDescription() {
094                    return description;
095            }
096    
097            /**
098             * @return the contributors
099             */
100            public Contributor[] getContributors() {
101                    return contributors;
102            }
103    
104            /**
105             * @return the main
106             */
107            public String getMain() {
108                    return main;
109            }
110    
111            /**
112             * @return the github
113             */
114            public String getGithub() {
115                    return github;
116            }
117    
118            /**
119             * @return the dependencies
120             */
121            public String[] getDependencies() {
122                    return dependencies;
123            }
124    
125            /**
126             * @return the license
127             */
128            public String getLicense() {
129                    return license;
130            }
131    
132            /**
133             * @return the engineVersion
134             */
135            public String getEngineVersion() {
136                    return engineVersion;
137            }
138    
139            /**
140             * Represents a contributor to the mod.
141             */
142            public class Contributor {
143                    /**
144                     * The name of the contributor.
145                     */
146                    private final String name;
147    
148                    /**
149                     * The email of the contributor.
150                     */
151                    private final String email;
152    
153                    public Contributor(String name, String email) {
154                            this.name = name;
155                            this.email = email;
156                    }
157            }
158    }
