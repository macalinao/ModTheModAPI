001    package com.modthemod.api.property;
002    
003    import com.modthemod.api.base.Base;
004    import com.modthemod.api.mod.Mod;
005    
006    /**
007     * Manages {@link Type}s.
008     */
009    public interface TypeManager {
010            /**
011             * Gets a {@link Matcher} from its string representation.
012             * 
013             * @param name
014             *            The name of the {@link Matcher}.
015             * @return The {@link Matcher} corresponding with the given name.
016             */
017            public Matcher getMatcher(String name);
018    
019            /**
020             * Gets a {@link Type} from its name.
021             * 
022             * @param name
023             *            The name of the {@link Type}.
024             * @return The {@link Type} corresponding with the given name.
025             */
026            public Type<?> getType(String name);
027    
028            /**
029             * Registers a {@link Primitive} with the {@link TypeManager}.
030             * 
031             * @param primitive
032             *            The {@link Primitive} to register.
033             */
034            public void registerPrimitive(Primitive<?> primitive);
035    
036            /**
037             * Registers a {@link Base} with the {@link TypeManager}.
038             * 
039             * @param base
040             *            The {@link Base} to register.
041             */
042            public void registerBase(Base base);
043    }
