001    package com.modthemod.api.property;
002    
003    /**
004     * Manages {@link Type}s.
005     */
006    public interface TypeManager {
007            /**
008             * Gets a {@link Type} from its name.
009             * 
010             * @param name
011             *            The name of the {@link Type}.
012             * @return The {@link Type} corresponding with the given name.
013             */
014            public Type<?> getType(String name);
015    
016            /**
017             * Registers a {@link Primitive} with the {@link TypeManager}.
018             * 
019             * @param primitive
020             *            The {@link Primitive} to register.
021             */
022            public void registerPrimitive(Primitive<?> primitive);
023    }
