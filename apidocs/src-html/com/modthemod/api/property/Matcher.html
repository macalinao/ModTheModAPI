001    package com.modthemod.api.property;
002    
003    import com.modthemod.api.ModTheMod;
004    
005    /**
006     * Matches {@link Type}s.
007     */
008    public abstract class Matcher {
009            /**
010             * Returns a {@link String} representation of this {@link Matcher}.
011             * 
012             * <p>
013             * This representation, when plugged back into {@link Matcher#valueOf}, must
014             * return this {@link Matcher} once again. This should fit the regex of
015             * /a-z/g unless it is a special type.
016             * </p>
017             * 
018             * @return A {@link String} representation of this {@link Matcher}.
019             */
020            public abstract String toString();
021    
022            /**
023             * Returns true if the {@link Matcher} matches the given {@link Type}.
024             * 
025             * @param other
026             *            The other {@link Type} to check against.
027             * @return True if the {@link Matcher} is of the given {@link Type}.
028             */
029            public abstract boolean is(Type<?> other);
030    
031            /**
032             * Gets a {@link Matcher} from its {@link String} representation.
033             * 
034             * @param name
035             *            The name of the {@link Matcher}.
036             * @return The {@link Matcher} corresponding with the {@link String}.
037             */
038            public static Matcher valueOf(String name) {
039                    return ModTheMod.getGame().getTypeManager().getMatcher(name);
040            }
041    }
