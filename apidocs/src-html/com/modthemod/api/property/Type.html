001    package com.modthemod.api.property;
002    
003    import com.modthemod.api.ModTheMod;
004    
005    /**
006     * Represents a type of variable or constant.
007     * 
008     * @param V
009     *            The type of the content of the variable.
010     */
011    public abstract class Type<T> extends Matcher {
012            /**
013             * Gets a {@link String} representation of this {@link Type}.
014             * 
015             * <p>
016             * This representation, when plugged back into {@link Type#valueOf}, must
017             * return this {@link Type} once again. This should fit the regex of /a-z/g
018             * unless it is a special type.
019             * </p>
020             * 
021             * @return A {@link String} representation of this {@link Type}.
022             */
023            public abstract String toString();
024    
025            /**
026             * Creates a new {@link Property} with the given value.
027             * 
028             * @param value
029             *            The value of the {@link Property}.
030             * @return The created {@link Property}.
031             */
032            public Property<T> create(T value) {
033                    return new Property<T>(value);
034            }
035    
036            /**
037             * Gets a {@link Type} from its name.
038             * 
039             * @param name
040             *            The name of the {@link Type}.
041             * @return The {@link Type} corresponding with the given name.
042             */
043            public static Type<?> valueOf(String name) {
044                    return ModTheMod.getGame().getTypeManager().getType(name);
045            }
046    }
