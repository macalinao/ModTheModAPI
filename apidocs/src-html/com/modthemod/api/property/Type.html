001    package com.modthemod.api.property;
002    
003    /**
004     * Represents a type of variable or constant.
005     * 
006     * @param V
007     *            The type of the content of the variable.
008     */
009    public interface Type<T> {
010            /**
011             * Returns true if the {@link Type} is of the given {@link Type}.
012             * 
013             * @param other
014             *            The other {@link Type} to check against.
015             * @return True if the {@link Type} is of the given {@link Type}.
016             */
017            public boolean is(Type<?> other);
018    }
