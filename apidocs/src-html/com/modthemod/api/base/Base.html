001    package com.modthemod.api.base;
002    
003    import java.io.Serializable;
004    import java.util.List;
005    import java.util.Map;
006    
007    import com.modthemod.api.ModTheMod;
008    import com.modthemod.api.entity.Entity;
009    import com.modthemod.api.entity.EntityInstantiationException;
010    import com.modthemod.api.event.EventListener;
011    import com.modthemod.api.event.EventType;
012    import com.modthemod.api.mod.Mod;
013    import com.modthemod.api.property.Property;
014    import com.modthemod.api.property.Type;
015    import com.modthemod.api.util.Method;
016    
017    /**
018     * Represents a base of an {@link Entity}.
019     * 
020     * <p>
021     * Bases are immutable. The only thing that is "changed" with bases are the
022     * event listeners, which can be added or removed at will.
023     * </p>
024     */
025    public final class Base extends Type<Entity> implements Serializable, Cloneable {
026            /**
027             * Serial version UID.
028             */
029            private static final long serialVersionUID = -6874978692908808758L;
030    
031            /**
032             * The name of the base.
033             */
034            private final String name;
035    
036            /**
037             * The {@link Mod} of the base.
038             */
039            private final Mod mod;
040    
041            /**
042             * The Map containing the properties of the base. Immutable.
043             */
044            final Map<String, Property<?>> properties;
045    
046            /**
047             * The {@link Map} containing the {@link Method}s of the {@link Base}.
048             * Immutable.
049             */
050            final Map<String, Method> methods;
051    
052            /**
053             * The {@link Instantiator} that runs on entities when they are created.
054             */
055            private Instantiator instantiator;
056    
057            /**
058             * The {@link EntityFinder} that finds entities of this type.
059             */
060            private EntityFinder entityFinder;
061    
062            /**
063             * Creates a new {@link Base}.
064             * 
065             * <p>
066             * Please do not use this constructor.
067             * 
068             * Instead, use {@link BaseBuilder#build}.
069             * </p>
070             * 
071             * @param name
072             *            The name of the {@link Base}.
073             * @param properties
074             *            The properties of the {@link Base}.
075             * @param methods
076             *            The methods of the {@link Base}.
077             * @param instantiator
078             *            The instantiator of the {@link Base}.
079             */
080            Base(String name, Map<String, Property<?>> properties,
081                            Map<String, Method> methods, Instantiator instantiator,
082                            EntityFinder entityFinder, Mod mod) {
083                    this.name = name;
084                    this.properties = properties;
085                    this.methods = methods;
086                    this.instantiator = instantiator;
087                    this.entityFinder = entityFinder;
088                    this.mod = mod;
089            }
090    
091            /**
092             * Gets the name of this {@link Base}.
093             * 
094             * @return The name of this {@link Base}.
095             */
096            public String getName() {
097                    return name;
098            }
099    
100            /**
101             * Gets the {@link Mod} associated with this {@link Base}.
102             * 
103             * @return The {@link Mod} associated with this {@link Base}.
104             */
105            public Mod getMod() {
106                    return mod;
107            }
108    
109            /**
110             * Gets the value of a property.
111             * 
112             * @param name
113             *            The name of the property.
114             * @return The {@link Type}.
115             */
116            public Property<?> getProperty(String name) {
117                    return properties.get(name);
118            }
119    
120            /**
121             * Adds an {@link EventListener} to listen to this {@link Base}.
122             * 
123             * @param eventType
124             *            The name of the {@link EventType} the listener listens to.
125             * @param listener
126             *            The {@link EventListener} to listen.
127             */
128            public void addEventListener(String eventType, EventListener listener) {
129                    EventType type = ModTheMod.getGame().getEventManager()
130                                    .getEvent(eventType);
131    
132                    if (type == null) {
133                            throw new IllegalArgumentException("Unknown event type '"
134                                            + eventType + "' given!");
135                    }
136    
137                    addEventListener(type, listener);
138            }
139    
140            /**
141             * Adds an {@link EventListener} to listen to this {@link Base}.
142             * 
143             * @param eventType
144             *            The {@link EventType} of the listener.
145             * @param listener
146             *            The {@link EventListener} to listen.
147             */
148            public void addEventListener(EventType eventType, EventListener listener) {
149                    eventType.addListener(listener);
150            }
151    
152            /**
153             * Returns true if the {@link Entity}s are instantiatable.
154             * 
155             * @return True
156             */
157            public boolean isInstantiatable() {
158                    return instantiator != null;
159            }
160    
161            /**
162             * Instantiates a new {@link Entity} based on this {@link Base}.
163             * 
164             * @param args
165             *            The arguments to instantiate the {@link Entity} with.
166             * @return A new {@link Entity} derived from this {@link Base}.
167             * @throws EntityInstantiationException
168             *             If the entity was not able to be instantiated
169             */
170            public Entity instantiate(Object... args)
171                            throws EntityInstantiationException {
172                    if (!isInstantiatable()) {
173                            throw new EntityInstantiationException(
174                                            "Cannot instantiate entities of Base '" + getName() + "'!");
175                    }
176    
177                    try {
178                            return instantiator.instantiate(args);
179                    } catch (Exception ex) {
180                            throw new EntityInstantiationException(
181                                            "Could not instantiate an entity of Base '" + getName()
182                                                            + "'!", ex);
183                    }
184            }
185    
186            /**
187             * Creates a new {@link BaseBuilder} based on this {@link Base}.
188             * 
189             * @return A new {@link BaseBuilder} based on this {@link Base}.
190             */
191            public BaseBuilder extend() {
192                    return new BaseBuilder("a"); // TODO think about how this will work
193            }
194    
195            /**
196             * Returns true if the {@link Base} is a child of the other {@link Base}.
197             * 
198             * @param other
199             *            The other {@link Base}.
200             * @return True if the {@link Base} is a child of the other {@link Base}.
201             */
202            public boolean isChildOf(Base other) {
203                    return false; // TODO
204            }
205    
206            /**
207             * Finds entities of this base.
208             * 
209             * @param query
210             * @return
211             */
212            public List<Entity> find(Map<String, Property<?>> query) {
213                    return entityFinder.find(query);
214            }
215    
216            @Override
217            public String toString() {
218                    return "(" + getName() + ")";
219            }
220    
221            @Override
222            public boolean is(Type<?> other) {
223                    if (!(other instanceof Base)) {
224                            return false;
225                    }
226    
227                    Base that = (Base) other;
228    
229                    // This is not that if that is a superset of this.
230                    if (!that.equals(this) || !that.isChildOf(this)) {
231                            return false;
232                    }
233    
234                    return true;
235            }
236    
237    }
