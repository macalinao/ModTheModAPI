001    package com.modthemod.api.base;
002    
003    import java.io.Serializable;
004    import java.util.Map;
005    
006    import com.modthemod.api.ModTheMod;
007    import com.modthemod.api.entity.Entity;
008    import com.modthemod.api.event.EventListener;
009    import com.modthemod.api.event.EventType;
010    import com.modthemod.api.mod.Mod;
011    import com.modthemod.api.property.Property;
012    import com.modthemod.api.property.Type;
013    import com.modthemod.api.util.Method;
014    
015    /**
016     * Represents a base of an {@link Entity}.
017     * 
018     * <p>
019     * Bases are immutable. The only thing that is "changed" with bases are the
020     * event listeners, which can be added or removed at will.
021     * </p>
022     */
023    public final class Base extends Type<Entity> implements Serializable, Cloneable {
024            /**
025             * Serial version UID.
026             */
027            private static final long serialVersionUID = -6874978692908808758L;
028    
029            /**
030             * The name of the base.
031             */
032            private final String name;
033    
034            /**
035             * The {@link Mod} of the base.
036             */
037            private final Mod mod;
038    
039            /**
040             * The Map containing the properties of the base. Immutable.
041             */
042            final Map<String, Property<?>> properties;
043    
044            /**
045             * The {@link Map} containing the {@link Method}s of the {@link Base}.
046             * Immutable.
047             */
048            final Map<String, Method> methods;
049    
050            /**
051             * The {@link Instantiator} that runs on entities when they are created.
052             */
053            private Instantiator instantiator;
054    
055            /**
056             * Creates a new {@link Base}.
057             * 
058             * <p>
059             * Please do not use this constructor.
060             * 
061             * Instead, use {@link BaseBuilder#build}.
062             * </p>
063             * 
064             * @param name
065             *            The name of the {@link Base}.
066             * @param properties
067             *            The properties of the {@link Base}.
068             * @param methods
069             *            The methods of the {@link Base}.
070             * @param instantiator
071             *            The instantiator of the {@link Base}.
072             */
073            Base(String name, Map<String, Property<?>> properties,
074                            Map<String, Method> methods, Instantiator instantiator, Mod mod) {
075                    this.name = name;
076                    this.properties = properties;
077                    this.methods = methods;
078                    this.instantiator = instantiator;
079                    this.mod = mod;
080            }
081    
082            /**
083             * Gets the name of this {@link Base}.
084             * 
085             * @return The name of this {@link Base}.
086             */
087            public String getName() {
088                    return name;
089            }
090    
091            /**
092             * Gets the {@link Mod} associated with this {@link Base}.
093             * 
094             * @return The {@link Mod} associated with this {@link Base}.
095             */
096            public Mod getMod() {
097                    return mod;
098            }
099    
100            /**
101             * Gets the value of a property.
102             * 
103             * @param name
104             *            The name of the property.
105             * @return The {@link Type}.
106             */
107            public Property<?> getProperty(String name) {
108                    return properties.get(name);
109            }
110    
111            /**
112             * Adds an {@link EventListener} to listen to this {@link Base}.
113             * 
114             * @param eventType
115             *            The name of the {@link EventType} the listener listens to.
116             * @param listener
117             *            The {@link EventListener} to listen.
118             */
119            public void addEventListener(String eventType, EventListener listener) {
120                    EventType type = ModTheMod.getGame().getEventManager()
121                                    .getEvent(eventType);
122    
123                    if (type == null) {
124                            throw new IllegalArgumentException("Unknown event type '"
125                                            + eventType + "' given!");
126                    }
127    
128                    addEventListener(type, listener);
129            }
130    
131            /**
132             * Adds an {@link EventListener} to listen to this {@link Base}.
133             * 
134             * @param eventType
135             *            The {@link EventType} of the listener.
136             * @param listener
137             *            The {@link EventListener} to listen.
138             */
139            public void addEventListener(EventType eventType, EventListener listener) {
140                    eventType.addListener(listener);
141            }
142    
143            /**
144             * Gets the {@link Instantiator} of the {@link Base}.
145             * 
146             * @return The {@link Instantiator} of the {@link Base}.
147             */
148            public Instantiator getInstantiator() {
149                    return instantiator;
150            }
151    
152            /**
153             * Creates a new {@link BaseBuilder} based on this {@link Base}.
154             * 
155             * @return A new {@link BaseBuilder} based on this {@link Base}.
156             */
157            public BaseBuilder extend() {
158                    return new BaseBuilder("a"); // TODO think about how this will work
159            }
160    
161            /**
162             * Instantiates a new {@link Entity} based on this {@link Base}.
163             * 
164             * @return A new {@link Entity} derived from this {@link Base}.
165             */
166            public Entity instantiate() {
167                    return instantiate(new Object());
168            }
169    
170            /**
171             * Instantiates a new {@link Entity} based on this {@link Base}.
172             * 
173             * @param args
174             *            The arguments to instantiate the {@link Entity} with.
175             * @return A new {@link Entity} derived from this {@link Base}.
176             */
177            public Entity instantiate(Object... args) {
178                    return ModTheMod.getGame().getEntityManager()
179                                    .instantiateEntity(this, args);
180            }
181    
182            /**
183             * Returns true if the {@link Base} is a child of the other {@link Base}.
184             * 
185             * @param other
186             *            The other {@link Base}.
187             * @return True if the {@link Base} is a child of the other {@link Base}.
188             */
189            public boolean isChildOf(Base other) {
190                    return false; // TODO
191            }
192    
193            @Override
194            public String toString() {
195                    return "(" + getName() + ")";
196            }
197    
198            @Override
199            public boolean is(Type<?> other) {
200                    if (!(other instanceof Base)) {
201                            return false;
202                    }
203    
204                    Base that = (Base) other;
205    
206                    // This is not that if that is a superset of this.
207                    if (!that.equals(this) || !that.isChildOf(this)) {
208                            return false;
209                    }
210    
211                    return true;
212            }
213    
214    }
