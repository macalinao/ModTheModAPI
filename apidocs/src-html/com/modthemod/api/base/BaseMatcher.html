001    package com.modthemod.api.base;
002    
003    import java.util.ArrayList;
004    import java.util.Arrays;
005    import java.util.List;
006    
007    import com.modthemod.api.property.Matcher;
008    import com.modthemod.api.property.Type;
009    
010    /**
011     * Represents a {@link Type} corresponding with {@link Base}s..
012     */
013    public class BaseMatcher extends Matcher {
014            /**
015             * The {@link Base}s that have to be matched.
016             */
017            private List<Base> bases;
018    
019            /**
020             * Constructor for a {@link BaseMatcher}.<strong>Note: Please use
021             * {@link Matcher#valueOf}.</strong>
022             * 
023             * @param bases
024             *            The bases that compose this {@link BaseMatcher}.
025             */
026            public BaseMatcher(Base... bases) {
027                    this.bases = new ArrayList<Base>(Arrays.asList(bases));
028            }
029    
030            @Override
031            public boolean is(Type<?> other) {
032                    if (!(other instanceof Base)) {
033                            return false;
034                    }
035    
036                    Base baseType = (Base) other;
037                    for (Base base : bases) {
038                            if (!base.is(baseType)) {
039                                    return false;
040                            }
041                    }
042    
043                    return true;
044            }
045    
046            @Override
047            public String toString() {
048                    StringBuilder builder = new StringBuilder("(");
049                    boolean first = true;
050                    for (Base base : bases) {
051                            if (!first) {
052                                    builder.append(",");
053                            }
054                            builder.append(base.getName());
055                    }
056                    builder.append(")");
057                    return builder.toString();
058            }
059    
060    }
